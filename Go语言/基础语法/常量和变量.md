# 常量和变量

## 常量

```go

// 基础常量声明
const pi = 3.14159              // 无类型常量
const height int = 180          // 有类型常量

// 批量常量声明
const (
    statusOK   = 200            // 无类型整数常量
    statusNot  = 404            // 无类型整数常量
    msg        = "Not Found"    // 无类型字符串常量
)

// iota常量生成器应用
const (
    a = iota  // 0
    b         // 1（自动继承a的值）
    c         // 2
    d = 5     // 5（重置）
    e         // 5（继续继承）
    f = iota  // 4（重新使用iota）
    g         // 5
)

// 使用iota实现位掩码
const (
    flagRead   = 1 << iota  // 1
    flagWrite               // 2
    flagExecute             // 4
)
```

## 变量

```go
// 标准变量声明
var age int                    // 声明未赋值，默认为0
var name string = "Tom"        // 声明并赋值
var (
    count  int
    price  float64 = 99.99
    active bool
)

// 批量赋值
var x, y, z int
x, y = 1, 2                    // 交换值
x, y = y, x                    // 交换变量

// 短变量声明（函数内使用）
message := "Hello, Go!"        // 自动推断类型
number := 42                   // 推断为int
PI := 3.14159                  // 推断为float64

// 多变量短声明
a, b := 10, 20                 // 声明并赋值
a, c := 30, 40                 // a已存在（重新赋值），c是新变量

// 空白标识符使用
_, err := someFunc()           // 忽略不需要的返回值
_, _ = age, name               // 编译器忽略未使用的变量
```
